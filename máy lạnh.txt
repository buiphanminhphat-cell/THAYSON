    <script>
        const powerToggle = document.getElementById('powerToggle');
        const powerText = document.getElementById('powerText');
        const switchDot = document.getElementById('switchDot');
        const tempBtn = document.getElementById('tempBtn');
        const humidBtn = document.getElementById('humidBtn');
        const mainValue = document.getElementById('mainValue');
        const mainUnit = document.getElementById('mainUnit');
        const mainLabel = document.getElementById('mainLabel');
        const gaugeRing = document.getElementById('gaugeRing');
        const dashboard = document.getElementById('dashboard');

        let isPowerOn = true;
        let isTemperatureSelected = true;

        // Dữ liệu giả lập
        const data = {
            temp: { value: 17, unit: '°', label: 'Nhiệt độ lý tưởng' },
            humid: { value: 65, unit: '%', label: 'Mức độ ẩm thoải mái' }
        };

        // Hàm cập nhật trạng thái Bật/Tắt nguồn
        function updatePowerState() {
            if (isPowerOn) {
                powerToggle.style.backgroundColor = '#10b981'; // Primary Green
                powerText.textContent = 'Bật';
                switchDot.classList.remove('translate-x-full'); // Reset switch dot position if needed
                switchDot.classList.add('translate-x-0');
                dashboard.classList.remove('opacity-50', 'pointer-events-none');
            } else {
                powerToggle.style.backgroundColor = '#9ca3af'; // Gray
                powerText.textContent = 'Tắt';
                switchDot.classList.remove('translate-x-0');
                switchDot.classList.add('translate-x-full');
                dashboard.classList.add('opacity-50', 'pointer-events-none');
            }
        }

        // Hàm cập nhật chế độ Nhiệt độ/Độ ẩm
        function updateMode() {
            if (isTemperatureSelected) {
                // Chế độ Nhiệt độ
                tempBtn.classList.add('bg-primary-green', 'text-white', 'shadow-md');
                humidBtn.classList.remove('bg-primary-green', 'text-white', 'shadow-md');
                humidBtn.classList.add('text-gray-500');

                mainValue.textContent = data.temp.value;
                mainUnit.textContent = data.temp.unit;
                mainLabel.textContent = data.temp.label;
                gaugeRing.style.background = `conic-gradient(#8b5cf6 0%, #f472b6 30%, #f97316 60%, #8b5cf6 100%)`;

            } else {
                // Chế độ Độ ẩm
                humidBtn.classList.add('bg-primary-green', 'text-white', 'shadow-md');
                tempBtn.classList.remove('bg-primary-green', 'text-white', 'shadow-md');
                tempBtn.classList.add('text-gray-500');

                mainValue.textContent = data.humid.value;
                mainUnit.textContent = data.humid.unit;
                mainLabel.textContent = data.humid.label;
                // Đổi gradient cho Độ ẩm (ví dụ: sang tông màu xanh dương/lục)
                gaugeRing.style.background = `conic-gradient(#38bdf8 0%, #60a5fa 30%, #10b981 60%, #38bdf8 100%)`;
            }
        }

        // Xử lý sự kiện
        powerToggle.addEventListener('click', () => {
            isPowerOn = !isPowerOn;
            updatePowerState();
        });

        tempBtn.addEventListener('click', () => {
            isTemperatureSelected = true;
            updateMode();
        });

        humidBtn.addEventListener('click', () => {
            isTemperatureSelected = false;
            updateMode();
        });

        // Khởi tạo trạng thái ban đầu
        window.onload = () => {
            updatePowerState();
            updateMode();
        };

    </script>
